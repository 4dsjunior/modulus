        /** Renderiza a segmentação aninhada de Modalidade -> Frequência -> Sexo */
        function renderModalityFrequency(data) {
            const container = document.getElementById('stats-modality-frequency');
            container.innerHTML = '';
            
            // Itera sobre as modalidades (nível superior)
            for (const modality in data) {
                const modData = data[modality];
                const totalModality = modData.total;

                // Adiciona a classe active/bold para o item selecionado
                const isActive = modality === selectedModality ? 'active' : '';

                const modalityDiv = document.createElement('div');
                modalityDiv.className = `border-b border-gray-200 pb-2 mb-3 px-2 py-1 modality-item ${isActive}`;
                modalityDiv.setAttribute('data-modality', modality); // Adiciona atributo para clique
                modalityDiv.innerHTML = `
                    <!-- Raleway SemiBold 20px (simulado) -->
                    <p class="text-[1.25rem] font-raleway font-semibold text-primary">${modality} 
                        <span class="text-base font-quicksand font-medium text-secondary">(Total: ${totalModality} Alunos)</span>
                    </p>
                    <ul class="ml-4 mt-1 space-y-1 text-secondary">
                        ${Object.keys(modData)
                            .filter(key => key !== 'total')
                            .sort() 
                            .map(freq => {
                                const freqData = modData[freq];
                                // CORREÇÃO: Substituindo ícones unicode por M e F
                                return `
                                    <li class="font-quicksand text-sm">
                                        <span class="font-medium text-primary">-> ${freq} Semana:</span> ${freqData.total} alunos
                                        <span class="text-xs ml-2">
                                            (M ${freqData.masc || 0}, F ${freqData.fem || 0})
                                        </span>
                                    </li>`;
                            }).join('')}
                    </ul>
                `;
                container.appendChild(modalityDiv);
            }
        }

        /** Cria ou atualiza um gráfico Chart.js (geral) */
        function updateChart(chartId, type, labels, data, title, backgroundColors, isRevenueMode, instanceRef) {
            const ctx = document.getElementById(chartId).getContext('2d');
            
            // Destrói a instância anterior se existir
            if (instanceRef && instanceRef.value) {
                instanceRef.value.destroy();
            }

            const newChartInstance = new Chart(ctx, {
                type: type,
                data: {
                    labels: labels,
                    datasets: [{
                        data: data,
                        backgroundColor: backgroundColors,
                        hoverOffset: 12, // Aumenta o hover para melhor UX
                        borderWidth: 0, // Borda zero para donut chart limpo
                        borderRadius: 5,
                        spacing: 2
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false, 
                    plugins: {
                        legend: {
                            display: true, 
                            position: 'bottom',
                            labels: {
                                font: {
                                    size: 14,
                                    family: 'Quicksand',
                                }
                            }
                        },
                        title: {
                            display: false
                        },
                        tooltip: {
                            callbacks: {
                                label: function(context) {
                                    let label = context.label || '';
                                    if (label) {
                                        label += ': ';
                                    }
                                    if (context.parsed !== null) {
                                        if (isRevenueMode) {
                                            label += formatCurrency(context.parsed);
                                        } else {
                                            label += context.parsed + ' alunos';
                                        }
                                    }
                                    return label;
                                }
                            },
                            bodyFont: {
                                family: 'Quicksand',
                                size: 14
                            },
                            titleFont: {
                                family: 'Quicksand',
                                size: 14
                            }
                        }
                    }
                }
            });

            instanceRef.value = newChartInstance;
        }

        /** Renderiza o gráfico principal (FATURAMENTO POTENCIAL) */
        function renderPrimaryModalityChart() {
            const labels = Object.keys(modalityRevenueCounts);

        }

        /** Renderiza o gráfico de quebra de frequência (CONTAGEM DE ALUNOS) */
        function renderFrequencyChart(modality) {
            const modData = modalityFrequencyCounts[modality];
            const frequencyData = Object.keys(modData).filter(key => key !== 'total');
            
            const labels = frequencyData.map(freq => `${freq} Semana`);
            const data = frequencyData.map(freq => modData[freq].total);
            const colors = frequencyData.map((_, index) => FREQUENCY_COLORS[index % FREQUENCY_COLORS.length]);
            
            // Renderiza no modo CONTAGEM (isRevenueMode = false)
            updateChart('modalityChart', 'doughnut', labels, data, `Frequência em ${modality}`, colors, false, ModalityChart);

            document.getElementById('chart-title').textContent = `Distribuição em ${modality} (Contagem de Alunos)`;
            document.getElementById('reset-chart-btn').classList.remove('hidden');
            document.getElementById('reset-chart-btn').textContent = 'Mostrar Faturamento Potencial';


            // Define o estado de seleção e re-renderiza a lista para adicionar o destaque
            selectedModality = modality;
            renderModalityFrequency(modalityFrequencyCounts);
        }

        async function approvePayment(paymentId, studentWhatsapp, amount) {
            await new Promise(r => setTimeout(r, 500)); 
            
            const approvedPayment = mockPendingPayments.find(p => p.id === paymentId);
            if (approvedPayment) {
                mockPendingPayments = mockPendingPayments.filter(p => p.id !== paymentId);
                
                mockStats.approvedPayments.push({
                    amount: parseFloat(amount),
                    validated_at: new Date().toISOString()
                });

                const student = mockStats.students.find(s => s.whatsapp === studentWhatsapp);
                if (student) {
                    // Simula a atualização da data de vencimento para +30 dias
                    const newDueDate = new Date();
                    newDueDate.setDate(newDueDate.getDate() + 30); 
                    student.due_date = newDueDate.toISOString().split('T')[0];
                }
                
                renderPaymentsList(mockPendingPayments);
                fetchAndRenderStatistics(); 
                showMessage('Pagamento APROVADO e mensalidade atualizada (MOCK)!', 'success');

            } else {
                showMessage('Erro (MOCK): Pagamento não encontrado na lista pendente.', 'error');
            }
        }

        async function registerNewStudent(formData) {
            await new Promise(r => setTimeout(r, 500)); 

            // Simulação de verificação de WhatsApp duplicado (MVP)
            const exists = mockStats.students.some(s => s.whatsapp === formData.whatsapp);
            if (exists) {
                showMessage(`Erro: Aluno com WhatsApp ${formData.whatsapp} já está cadastrado.`, 'error');
                return false;
            }

            const newStudent = {
                name: formData.name,
                whatsapp: formData.whatsapp,
                due_date: formData.dueDate,
                status: "active",
                modality: formData.modality,
                gender: formData.gender,
                classes_per_week: formData.classesPerWeek,
                monthly_fee: parseFloat(formData.monthlyFee),
            };

            mockStats.students.push(newStudent);
            fetchAndRenderStatistics(); 
            closeRegistrationModal();
            showMessage(`Aluno ${formData.name} cadastrado com sucesso!`, 'success');
            return true;
        }


        async function fetchAndRenderStatistics() {
            await new Promise(r => setTimeout(r, 500)); 

            document.getElementById('statistics-section').classList.remove('hidden');
            
            const now = new Date();
            const currentYear = now.getFullYear();
            const currentMonth = now.getMonth();
            
            const startOfCurrentMonth = new Date(currentYear, currentMonth, 1);
            const startOfNextMonth = new Date(currentYear, currentMonth + 1, 1);

            // --- 1. Cálculo de Pagamentos Aprovados ---
            let annualRevenue = 0;
            let monthlyReceived = 0;
            
            mockStats.approvedPayments.forEach(payment => {
                const validatedDate = new Date(payment.validated_at);

                if (validatedDate.getFullYear() === currentYear) {
                    annualRevenue += payment.amount;
                }

                if (validatedDate >= startOfCurrentMonth && validatedDate < startOfNextMonth) {
                    monthlyReceived += payment.amount;
                }
            });


            // --- 2. Cálculo de Alunos e Segmentação ---
            let totalStudents = 0;
            let monthlyExpected = 0;
            let nextMonthForecast = 0;
            
            modalityFrequencyCounts = {}; 
            modalityRevenueCounts = {}; 

            mockStats.students.forEach(student => {
                const monthlyFee = student.monthly_fee || 100; 

                if (student.status === 'active') {
                    totalStudents++;
                    
                    // Segmentação Aninhada (Contagem)
                    const mod = student.modality || 'Não Informado';
                    const freq = student.classes_per_week || 'Não Informado';
                    const genderNormalized = student.gender && student.gender.toLowerCase().startsWith('m') ? 'masc' : 'fem';
                    
                    if (!modalityFrequencyCounts[mod]) {
                        modalityFrequencyCounts[mod] = { total: 0 };
                    }
                    if (!modalityFrequencyCounts[mod][freq]) {
                        modalityFrequencyCounts[mod][freq] = { masc: 0, fem: 0, total: 0 };
                    }
                    
                    modalityFrequencyCounts[mod].total += 1;
                    modalityFrequencyCounts[mod][freq].total += 1;
                    modalityFrequencyCounts[mod][freq][genderNormalized] += 1;
                    
                    // Faturamento Potencial (Receita)
                    modalityRevenueCounts[mod] = (modalityRevenueCounts[mod] || 0) + monthlyFee;

                    // Previsão Total
                    nextMonthForecast += monthlyFee;

                    // Esperado Mês Corrente
                    const dueDate = new Date(student.due_date);
                    if (dueDate >= startOfCurrentMonth && dueDate < startOfNextMonth) {
                        monthlyExpected += monthlyFee;
                    }
                }
            });
            
            // --- 3. Renderização dos Resultados ---

            // Faturamento
            document.getElementById('annual-revenue').textContent = formatCurrency(annualRevenue);
            document.getElementById('next-month-forecast').textContent = formatCurrency(nextMonthForecast);
            document.getElementById('total-students').textContent = totalStudents.toString();

            // Status Mensal
            const monthlyMissing = monthlyExpected - monthlyReceived;
            document.getElementById('monthly-expected').textContent = formatCurrency(monthlyExpected);
            document.getElementById('monthly-received').textContent = formatCurrency(monthlyReceived);
            document.getElementById('monthly-missing').textContent = formatCurrency(Math.max(0, monthlyMissing));

            // Segmentação (Lista Hierárquica e Gráficos)
            renderModalityFrequency(modalityFrequencyCounts); 
            
            // Mantém o estado do gráfico de Contagem E a sincronização do destaque na lista
            if (selectedModality && modalityFrequencyCounts[selectedModality]) {
                renderFrequencyChart(selectedModality);
            } else {
                renderPrimaryModalityChart(); // Padrão: mostra Faturamento
            }
            
            // FIX ADICIONAL: Garante que, se o selectedModality for nulo, nada fique ativo.
            if (!selectedModality) {
                document.querySelectorAll('.modality-item').forEach(item => item.classList.remove('active'));
            }
        }
